Schema жј”еҢ–

Apache Foryв„ў еңЁ Compatible жЁЎејҸдёӢж”ҜжҢҒ schema жј”еҢ–пјҢе…Ғи®ёеәҸеҲ—еҢ–е’ҢеҸҚеәҸеҲ—еҢ–еҸҢж–№жӢҘжңүдёҚеҗҢзҡ„зұ»еһӢе®ҡд№үгҖӮ

Compatible жЁЎејҸвҖӢ

дҪҝз”Ё compatible(true) еҗҜз”Ё schema жј”еҢ–пјҡ

#include "fory/serialization/fory.h"

using namespace fory::serialization;

// зүҲжң¬ 1пјҡеҺҹе§Ӣ schema
struct PersonV1 {
  std::string name;
  int32_t age;
};
FORY_STRUCT(PersonV1, name, age);

// зүҲжң¬ 2пјҡж·»еҠ  email еӯ—ж®ө
struct PersonV2 {
  std::string name;
  int32_t age;
  std::string email;  // ж–°еӯ—ж®ө
};
FORY_STRUCT(PersonV2, name, age, email);

int main() {
  // дёәжҜҸдёӘ schema зүҲжң¬еҲӣе»әеҚ•зӢ¬зҡ„ Fory е®һдҫӢ
  auto fory_v1 = Fory::builder()
      .compatible(true)  // еҗҜз”Ё schema жј”еҢ–
      .xlang(true)
      .build();

  auto fory_v2 = Fory::builder()
      .compatible(true)
      .xlang(true)
      .build();

  // дҪҝз”ЁзӣёеҗҢзҡ„зұ»еһӢ ID иҝӣиЎҢ schema жј”еҢ–
  constexpr uint32_t PERSON_TYPE_ID = 100;
  fory_v1.register_struct<PersonV1>(PERSON_TYPE_ID);
  fory_v2.register_struct<PersonV2>(PERSON_TYPE_ID);

  // дҪҝз”Ё V1 еәҸеҲ—еҢ–
  PersonV1 v1{"Alice", 30};
  auto bytes = fory_v1.serialize(v1).value();

  // еҸҚеәҸеҲ—еҢ–дёә V2 - email иҺ·еҫ—й»ҳи®ӨеҖјпјҲз©әеӯ—з¬ҰдёІпјү
  auto v2 = fory_v2.deserialize<PersonV2>(bytes).value();
  assert(v2.name == "Alice");
  assert(v2.age == 30);
  assert(v2.email == "");  // зјәеӨұеӯ—ж®өзҡ„й»ҳи®ӨеҖј

  return 0;
}

Schema жј”еҢ–зү№жҖ§вҖӢ

Compatible жЁЎејҸж”ҜжҢҒд»ҘдёӢ schema еҸҳжӣҙпјҡ

еҸҳжӣҙзұ»еһӢж”ҜжҢҒиЎҢдёә
ж·»еҠ ж–°еӯ—ж®өвң…зјәеӨұеӯ—ж®өдҪҝз”Ёй»ҳи®ӨеҖј
еҲ йҷӨеӯ—ж®өвң…йўқеӨ–еӯ—ж®өиў«и·іиҝҮ
йҮҚжҺ’еӯ—ж®өйЎәеәҸвң…жҢүеҗҚз§°еҢ№й…Қеӯ—ж®өпјҢиҖҢйқһдҪҚзҪ®
жӣҙж”№еҸҜз©әжҖ§вң…T вҶ” std::optional<T>
жӣҙж”№еӯ—ж®өзұ»еһӢвқҢзұ»еһӢеҝ…йЎ»е…је®№
йҮҚе‘ҪеҗҚеӯ—ж®өвқҢеӯ—ж®өеҗҚеҝ…йЎ»еҢ№й…ҚпјҲеҢәеҲҶеӨ§е°ҸеҶҷпјү

ж·»еҠ еӯ—ж®өпјҲеҗ‘еҗҺе…је®№пјүвҖӢ

еҪ“дҪҝз”Ёе…·жңүйўқеӨ–еӯ—ж®өзҡ„ж–° schema еҸҚеәҸеҲ—еҢ–ж—§ж•°жҚ®ж—¶пјҡ

// ж—§ schema (V1)
struct ProductV1 {
  std::string name;
  double price;
};
FORY_STRUCT(ProductV1, name, price);

// ж–° schema (V2) еёҰжңүйўқеӨ–еӯ—ж®ө
struct ProductV2 {
  std::string name;
  double price;
  std::vector<std::string> tags;       // ж–°еӯ—ж®ө
  std::map<std::string, std::string> attributes;  // ж–°еӯ—ж®ө
};
FORY_STRUCT(ProductV2, name, price, tags, attributes);

// еәҸеҲ—еҢ– V1
ProductV1 v1{"Laptop", 999.99};
auto bytes = fory_v1.serialize(v1).value();

// еҸҚеәҸеҲ—еҢ–дёә V2
auto v2 = fory_v2.deserialize<ProductV2>(bytes).value();
assert(v2.name == "Laptop");
assert(v2.price == 999.99);
assert(v2.tags.empty());        // й»ҳи®Өпјҡз©ә vector
assert(v2.attributes.empty());  // й»ҳи®Өпјҡз©ә map

еҲ йҷӨеӯ—ж®өпјҲеҗ‘еүҚе…је®№пјүвҖӢ

еҪ“дҪҝз”Ёе…·жңүиҫғе°‘еӯ—ж®өзҡ„ж—§ schema еҸҚеәҸеҲ—еҢ–ж–°ж•°жҚ®ж—¶пјҡ

// е®Ңж•ҙ schema
struct UserFull {
  int64_t id;
  std::string username;
  std::string email;
  std::string password_hash;
  int32_t login_count;
};
FORY_STRUCT(UserFull, id, username, email, password_hash, login_count);

// зІҫз®Җ schemaпјҲеҲ йҷӨдәҶ 3 дёӘеӯ—ж®өпјү
struct UserMinimal {
  int64_t id;
  std::string username;
};
FORY_STRUCT(UserMinimal, id, username);

// еәҸеҲ—еҢ–е®Ңж•ҙзүҲжң¬
UserFull full{12345, "johndoe", "john@example.com", "hash123", 42};
auto bytes = fory_full.serialize(full).value();

// еҸҚеәҸеҲ—еҢ–дёәзІҫз®ҖзүҲжң¬ - йўқеӨ–еӯ—ж®өиў«и·іиҝҮ
auto minimal = fory_minimal.deserialize<UserMinimal>(bytes).value();
assert(minimal.id == 12345);
assert(minimal.username == "johndoe");
// emailгҖҒpassword_hashгҖҒlogin_count иў«и·іиҝҮ

еӯ—ж®өйҮҚжҺ’вҖӢ

еңЁ compatible жЁЎејҸдёӢпјҢеӯ—ж®өжҢүеҗҚз§°еҢ№й…ҚпјҢиҖҢйқһдҪҚзҪ®пјҡ

// еҺҹе§Ӣеӯ—ж®өйЎәеәҸ
struct ConfigOriginal {
  std::string host;
  int32_t port;
  bool enable_ssl;
  std::string protocol;
};
FORY_STRUCT(ConfigOriginal, host, port, enable_ssl, protocol);

// йҮҚжҺ’еҗҺзҡ„еӯ—ж®ө
struct ConfigReordered {
  bool enable_ssl;      // з§»еҲ°з¬¬дёҖдҪҚ
  std::string protocol; // з§»еҲ°з¬¬дәҢдҪҚ
  std::string host;     // з§»еҲ°з¬¬дёүдҪҚ
  int32_t port;         // з§»еҲ°жңҖеҗҺ
};
FORY_STRUCT(ConfigReordered, enable_ssl, protocol, host, port);

// дҪҝз”ЁеҺҹе§ӢйЎәеәҸеәҸеҲ—еҢ–
ConfigOriginal orig{"localhost", 8080, true, "https"};
auto bytes = fory_orig.serialize(orig).value();

// дҪҝз”ЁдёҚеҗҢеӯ—ж®өйЎәеәҸеҸҚеәҸеҲ—еҢ– - жӯЈеёёе·ҘдҪң
auto reordered = fory_reord.deserialize<ConfigReordered>(bytes).value();
assert(reordered.host == "localhost");
assert(reordered.port == 8080);
assert(reordered.enable_ssl == true);
assert(reordered.protocol == "https");

еөҢеҘ—з»“жһ„дҪ“жј”еҢ–вҖӢ

Schema жј”еҢ–йҖ’еҪ’ж”ҜжҢҒеөҢеҘ—з»“жһ„дҪ“пјҡ

// V1 Address
struct AddressV1 {
  std::string street;
  std::string city;
};
FORY_STRUCT(AddressV1, street, city);

// V2 Address еёҰжңүж–°еӯ—ж®ө
struct AddressV2 {
  std::string street;
  std::string city;
  std::string country;  // ж–°еӯ—ж®ө
  std::string zipcode;  // ж–°еӯ—ж®ө
};
FORY_STRUCT(AddressV2, street, city, country, zipcode);

// V1 Employee дҪҝз”Ё V1 Address
struct EmployeeV1 {
  std::string name;
  AddressV1 home_address;
};
FORY_STRUCT(EmployeeV1, name, home_address);

// V2 Employee дҪҝз”Ё V2 Address е’Ңж–°еӯ—ж®ө
struct EmployeeV2 {
  std::string name;
  AddressV2 home_address;  // еөҢеҘ—з»“жһ„дҪ“е·Іжј”еҢ–
  std::string employee_id; // ж–°еӯ—ж®ө
};
FORY_STRUCT(EmployeeV2, name, home_address, employee_id);

// дҪҝз”ЁзӣёеҗҢ ID жіЁеҶҢзұ»еһӢ
constexpr uint32_t ADDRESS_TYPE_ID = 100;
constexpr uint32_t EMPLOYEE_TYPE_ID = 101;

fory_v1.register_struct<AddressV1>(ADDRESS_TYPE_ID);
fory_v1.register_struct<EmployeeV1>(EMPLOYEE_TYPE_ID);
fory_v2.register_struct<AddressV2>(ADDRESS_TYPE_ID);
fory_v2.register_struct<EmployeeV2>(EMPLOYEE_TYPE_ID);

// еәҸеҲ—еҢ– V1
EmployeeV1 emp_v1{"Jane Doe", {"123 Main St", "NYC"}};
auto bytes = fory_v1.serialize(emp_v1).value();

// еҸҚеәҸеҲ—еҢ–дёә V2
auto emp_v2 = fory_v2.deserialize<EmployeeV2>(bytes).value();
assert(emp_v2.name == "Jane Doe");
assert(emp_v2.home_address.street == "123 Main St");
assert(emp_v2.home_address.city == "NYC");
assert(emp_v2.home_address.country == "");  // й»ҳи®ӨеҖј
assert(emp_v2.home_address.zipcode == "");  // й»ҳи®ӨеҖј
assert(emp_v2.employee_id == "");           // й»ҳи®ӨеҖј

SchemaConsistent жЁЎејҸпјҲй»ҳи®ӨпјүвҖӢ

дёҚдҪҝз”Ё compatible жЁЎејҸж—¶пјҢschema еҝ…йЎ»е®Ңе…ЁеҢ№й…Қпјҡ

// дёҘж јжЁЎејҸпјҲй»ҳи®Өпјү
auto fory = Fory::builder()
    .compatible(false)  // й»ҳи®Өпјҡschema еҝ…йЎ»еҢ№й…Қ
    .xlang(true)
    .build();

// еәҸеҲ—еҢ–/еҸҚеәҸеҲ—еҢ–иҰҒжұӮзӣёеҗҢзҡ„ schema
// Schema дёҚеҢ№й…ҚеҸҜиғҪеҜјиҮҙй”ҷиҜҜжҲ–жңӘе®ҡд№үиЎҢдёә

дҪ•ж—¶дҪҝз”Ё SchemaConsistent жЁЎејҸпјҡ

  • Schema дҝқиҜҒеҢ№й…ҚпјҲзӣёеҗҢзҡ„дәҢиҝӣеҲ¶зүҲжң¬пјү
  • йңҖиҰҒжңҖй«ҳжҖ§иғҪпјҲжӣҙе°‘зҡ„е…ғж•°жҚ®ејҖй”Җпјү
  • жӮЁеҗҢж—¶жҺ§еҲ¶еәҸеҲ—еҢ–е’ҢеҸҚеәҸеҲ—еҢ–

дҪ•ж—¶дҪҝз”Ё Compatible жЁЎејҸпјҡ

  • Schema еҸҜиғҪзӢ¬з«Ӣжј”еҢ–
  • йңҖиҰҒи·ЁзүҲжң¬е…је®№жҖ§
  • дёҚеҗҢжңҚеҠЎеҸҜиғҪжңүдёҚеҗҢзҡ„ schema зүҲжң¬

зұ»еһӢ ID иҰҒжұӮвҖӢ

иҰҒдҪҝ schema жј”еҢ–жӯЈеёёе·ҘдҪңпјҡ

  1. зӣёеҗҢзұ»еһӢ IDпјҡеҗҢдёҖз»“жһ„дҪ“зҡ„дёҚеҗҢзүҲжң¬еҝ…йЎ»дҪҝз”ЁзӣёеҗҢзҡ„зұ»еһӢ ID
  2. дёҖиҮҙзҡ„ IDпјҡзұ»еһӢ ID еҝ…йЎ»еңЁжүҖжңү Fory е®һдҫӢдёӯдҝқжҢҒдёҖиҮҙ
  3. жіЁеҶҢжүҖжңүзүҲжң¬пјҡжҜҸдёӘ Fory е®һдҫӢжіЁеҶҢиҮӘе·ұзҡ„з»“жһ„дҪ“зүҲжң¬
constexpr uint32_t PERSON_TYPE_ID = 100;

// е®һдҫӢ 1 дҪҝз”Ё PersonV1
fory_v1.register_struct<PersonV1>(PERSON_TYPE_ID);

// е®һдҫӢ 2 дҪҝз”Ё PersonV2
fory_v2.register_struct<PersonV2>(PERSON_TYPE_ID);

// зӣёеҗҢзҡ„зұ»еһӢ ID еҗҜз”Ё schema жј”еҢ–

жңҖдҪіе®һи·өвҖӢ

1. дёәжј”еҢ–еҒҡи§„еҲ’вҖӢ

и®ҫи®Ў schema ж—¶иҖғиҷ‘жңӘжқҘзҡ„еҸҳжӣҙпјҡ

// еҘҪзҡ„еҒҡжі•пјҡеҜ№еҸҜиғҪеҲ йҷӨзҡ„еӯ—ж®өдҪҝз”Ё optional
struct Config {
  std::string host;
  int32_t port;
  std::optional<std::string> deprecated_field;  // д»ҘеҗҺеҸҜд»ҘеҲ йҷӨ
};

2. дҪҝз”Ёжңүж„Ҹд№үзҡ„й»ҳи®ӨеҖјвҖӢ

иҖғиҷ‘ж–°еӯ—ж®өдҪҝз”Ёд»Җд№Ҳй»ҳи®ӨеҖјжүҚжңүж„Ҹд№үпјҡ

struct Settings {
  int32_t timeout_ms;      // й»ҳи®Өпјҡ0пјҲеҸҜиғҪйңҖиҰҒдёҖдёӘеҗҲзҗҶзҡ„й»ҳи®ӨеҖјпјү
  bool enabled;            // й»ҳи®Өпјҡfalse
  std::string mode;        // й»ҳи®Өпјҡ""пјҲеҸҜиғҪйңҖиҰҒ "default"пјү
};

и·ҹиёӘ schema еҸҳжӣҙд»Ҙдҫҝи°ғиҜ•пјҡ

// V2пјҡж·»еҠ  email еӯ—ж®өпјҲ2024-02-01пјү
// V3пјҡж·»еҠ  phoneгҖҒaddress еӯ—ж®өпјҲ2024-03-01пјү

4. жөӢиҜ•жј”еҢ–и·Ҝеҫ„вҖӢ

жөӢиҜ•еҚҮзә§е’ҢйҷҚзә§еңәжҷҜпјҡ

// жөӢиҜ• V1 -> V2
// жөӢиҜ• V2 -> V1
// жөӢиҜ• V1 -> V3
// жөӢиҜ• V3 -> V1

и·ЁиҜӯиЁҖ Schema жј”еҢ–вҖӢ

дҪҝз”Ё xlang жЁЎејҸж—¶пјҢschema жј”еҢ–и·ЁиҜӯиЁҖе·ҘдҪңпјҡ

// дҪҝз”Ё compatible жЁЎејҸзҡ„ C++

    .compatible(true)
    .xlang(true)
    .build();
// дҪҝз”Ё compatible жЁЎејҸзҡ„ Java
