иҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁ

иҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁе…Ғи®ёдҪ зІҫзЎ®жҺ§еҲ¶зұ»еһӢзҡ„еәҸеҲ—еҢ–дёҺеҸҚеәҸеҲ—еҢ–иҝҮзЁӢгҖӮиҝҷйқһеёёйҖӮеҗҲйңҖиҰҒзү№ж®ҠеӨ„зҗҶгҖҒйўқеӨ–дјҳеҢ–жҲ–и·ЁиҜӯиЁҖе…је®№зҡ„зұ»еһӢгҖӮ

дҪ•ж—¶дҪҝз”ЁиҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁвҖӢ

  • зү№ж®Ҡзј–з Ғж јејҸпјҡзұ»еһӢйңҖиҰҒеӣәе®ҡжҲ–е®ҡеҲ¶зҡ„дәҢиҝӣеҲ¶ж јејҸ
  • з¬¬дёүж–№зұ»еһӢпјҡзұ»еһӢжқҘиҮӘеӨ–йғЁеә“пјҢFory ж— жі•иҮӘеҠЁеӨ„зҗҶ
  • жҖ§иғҪдјҳеҢ–пјҡдҪ иғҪжҜ”й»ҳи®Өзҡ„еҸҚе°„ж–№жЎҲе®һзҺ°жӣҙй«ҳж•Ҳзҡ„зј–з Ғ
  • и·ЁиҜӯиЁҖе…је®№пјҡдҪ йңҖиҰҒзІҫзЎ®жҺ§еҲ¶дәҢиҝӣеҲ¶еёғеұҖжқҘдҝқиҜҒдә’ж“ҚдҪң

ExtensionSerializer жҺҘеҸЈвҖӢ

иҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁйңҖиҰҒе®һзҺ° ExtensionSerializer жҺҘеҸЈпјҡ

type ExtensionSerializer interface {
    // WriteData е°ҶеҖјеҶҷе…Ҙзј“еҶІеҢәгҖӮ
    // иҝҷйҮҢеҸӘиҙҹиҙЈеҶҷе…Ҙж•°жҚ®жң¬иә«пјҢзұ»еһӢдҝЎжҒҜе’Ңеј•з”Ёз”ұ Fory еӨ„зҗҶгҖӮ
    // дҪҝз”Ё ctx.Buffer() иҺ·еҸ– ByteBufferгҖӮ
    // дҪҝз”Ё ctx.SetError() жҠҘе‘Ҡй”ҷиҜҜгҖӮ
    WriteData(ctx *WriteContext, value reflect.Value)

    // ReadData д»Һзј“еҶІеҢәиҜ»еҸ–еҖје№¶еҶҷе…ҘеҲ°жҸҗдҫӣзҡ„зӣ®ж ҮдёӯгҖӮ
    // иҝҷйҮҢеҸӘиҙҹиҙЈиҜ»еҸ–ж•°жҚ®жң¬иә«пјҢзұ»еһӢдҝЎжҒҜе’Ңеј•з”Ёз”ұ Fory еӨ„зҗҶгҖӮ
    // дҪҝз”Ё ctx.Buffer() иҺ·еҸ– ByteBufferгҖӮ
    // дҪҝз”Ё ctx.SetError() жҠҘе‘Ҡй”ҷиҜҜгҖӮ
    ReadData(ctx *ReadContext, value reflect.Value)
}

еҹәзЎҖзӨәдҫӢвҖӢ

дёӢйқўжҳҜдёҖдёӘд»…еҢ…еҗ«ж•ҙж•°еӯ—ж®өзҡ„з®ҖеҚ•иҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁпјҡ

import (
    "reflect"
    "github.com/apache/fory/go/fory"
)

type MyExt struct {
    Id int32
}

type MyExtSerializer struct{}

func (s *MyExtSerializer) WriteData(ctx *fory.WriteContext, value reflect.Value) {
    myExt := value.Interface().(MyExt)
    ctx.Buffer().WriteVarint32(myExt.Id)
}

func (s *MyExtSerializer) ReadData(ctx *fory.ReadContext, value reflect.Value) {
    id := ctx.Buffer().ReadVarint32(ctx.Err())
    value.Set(reflect.ValueOf(MyExt{Id: id}))
}

// жіЁеҶҢиҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁ
f := fory.New()
err := f.RegisterExtension(MyExt{}, 100, &MyExtSerializer{})

Context ж–№жі•вҖӢ

WriteContext е’Ң ReadContext жҸҗдҫӣдәҶи®ҝй—®еәҸеҲ—еҢ–иө„жәҗзҡ„иғҪеҠӣпјҡ

ж–№жі•иҜҙжҳҺ
Buffer()иҝ”еӣһз”ЁдәҺиҜ»еҶҷзҡ„ *ByteBuffer
Err()иҝ”еӣһ *ErrorпјҢз”ЁдәҺе»¶иҝҹй”ҷиҜҜжЈҖжҹҘ
SetError(err)еңЁдёҠдёӢж–Үдёӯи®°еҪ•й”ҷиҜҜ
HasError()еҲӨж–ӯдёҠдёӢж–ҮдёӯжҳҜеҗҰе·Іи®°еҪ•й”ҷиҜҜ
TypeResolver()иҝ”еӣһеөҢеҘ—зұ»еһӢдҪҝз”Ёзҡ„зұ»еһӢи§ЈжһҗеҷЁ
RefResolver()иҝ”еӣһеј•з”Ёи·ҹиёӘдҪҝз”Ёзҡ„и§ЈжһҗеҷЁ

ByteBuffer ж–№жі•вҖӢ

ByteBuffer жҸҗдҫӣдәҶдёҖз»„иҜ»еҸ–е’ҢеҶҷе…ҘеҺҹе§Ӣзұ»еһӢзҡ„ж–№жі•гҖӮ

еҶҷе…Ҙж–№жі•вҖӢ

ж–№жі•иҜҙжҳҺ
WriteBool(v bool)еҶҷе…Ҙеёғе°”еҖј
WriteInt8(v int8)еҶҷе…Ҙжңүз¬ҰеҸ· 8 дҪҚж•ҙж•°
WriteInt16(v int16)еҶҷе…Ҙжңүз¬ҰеҸ· 16 дҪҚж•ҙж•°
WriteInt32(v int32)еҶҷе…Ҙжңүз¬ҰеҸ· 32 дҪҚж•ҙж•°
WriteInt64(v int64)еҶҷе…Ҙжңүз¬ҰеҸ· 64 дҪҚж•ҙж•°
WriteFloat32(v float32)еҶҷе…Ҙ 32 дҪҚжө®зӮ№ж•°
WriteFloat64(v float64)еҶҷе…Ҙ 64 дҪҚжө®зӮ№ж•°
WriteVarint32(v int32)еҶҷе…ҘеҸҳй•ҝжңүз¬ҰеҸ· 32 дҪҚж•ҙж•°
WriteVarint64(v int64)еҶҷе…ҘеҸҳй•ҝжңүз¬ҰеҸ· 64 дҪҚж•ҙж•°
WriteBinary(data []byte)еҶҷе…ҘеҺҹе§Ӣеӯ—иҠӮ

иҜ»еҸ–ж–№жі•вҖӢ

жүҖжңүиҜ»еҸ–ж–№жі•йғҪжҺҘж”¶дёҖдёӘ *Error еҸӮж•°пјҢз”ЁдәҺе»¶иҝҹй”ҷиҜҜжЈҖжҹҘпјҡ

ж–№жі•иҜҙжҳҺ
ReadBool(err *Error) boolиҜ»еҸ–еёғе°”еҖј
ReadInt8(err *Error) int8иҜ»еҸ–жңүз¬ҰеҸ· 8 дҪҚж•ҙж•°
ReadInt16(err *Error) int16иҜ»еҸ–жңүз¬ҰеҸ· 16 дҪҚж•ҙж•°
ReadInt32(err *Error) int32иҜ»еҸ–жңүз¬ҰеҸ· 32 дҪҚж•ҙж•°
ReadInt64(err *Error) int64иҜ»еҸ–жңүз¬ҰеҸ· 64 дҪҚж•ҙж•°
ReadFloat32(err *Error) float32иҜ»еҸ– 32 дҪҚжө®зӮ№ж•°
ReadFloat64(err *Error) float64иҜ»еҸ– 64 дҪҚжө®зӮ№ж•°
ReadVarint32(err *Error) int32иҜ»еҸ–еҸҳй•ҝжңүз¬ҰеҸ· 32 дҪҚж•ҙж•°
ReadVarint64(err *Error) int64иҜ»еҸ–еҸҳй•ҝжңүз¬ҰеҸ· 64 дҪҚж•ҙж•°
ReadBinary(length int, err *Error) []byteиҜ»еҸ–жҢҮе®ҡй•ҝеәҰзҡ„еҺҹе§Ӣеӯ—иҠӮ

еӨҚжқӮзұ»еһӢзӨәдҫӢвҖӢ

дёӢйқўеұ•зӨәдёҖдёӘеҢ…еҗ«еӨҡдёӘеӯ—ж®өзҡ„иҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁпјҡ

type Point3D struct {
    X, Y, Z float64
    Label   string
}

type Point3DSerializer struct{}

func (s *Point3DSerializer) WriteData(ctx *fory.WriteContext, value reflect.Value) {
    p := value.Interface().(Point3D)
    buf := ctx.Buffer()
    buf.WriteFloat64(p.X)
    buf.WriteFloat64(p.Y)
    buf.WriteFloat64(p.Z)
    // еӯ—з¬ҰдёІеҶҷдёәвҖңй•ҝеәҰ + еҶ…е®№еӯ—иҠӮвҖқ
    labelBytes := []byte(p.Label)
    buf.WriteVarint32(int32(len(labelBytes)))
    buf.WriteBinary(labelBytes)
}

func (s *Point3DSerializer) ReadData(ctx *fory.ReadContext, value reflect.Value) {
    buf := ctx.Buffer()
    err := ctx.Err()
    x := buf.ReadFloat64(err)
    y := buf.ReadFloat64(err)
    z := buf.ReadFloat64(err)
    labelLen := buf.ReadVarint32(err)
    labelBytes := buf.ReadBinary(int(labelLen), err)
    value.Set(reflect.ValueOf(Point3D{
        X:     x,
        Y:     y,
        Z:     z,
        Label: string(labelBytes),
    }))
}

f := fory.New()
f.RegisterExtension(Point3D{}, 101, &Point3DSerializer{})

еӨ„зҗҶжҢҮй’ҲвҖӢ

еҰӮжһңзұ»еһӢдёӯеҢ…еҗ«жҢҮй’ҲпјҢйңҖиҰҒжҳҫејҸеӨ„зҗҶ nilпјҡ

type OptionalValue struct {
    Value *int64
}

type OptionalValueSerializer struct{}

func (s *OptionalValueSerializer) WriteData(ctx *fory.WriteContext, value reflect.Value) {
    ov := value.Interface().(OptionalValue)
    buf := ctx.Buffer()
    if ov.Value == nil {
        buf.WriteBool(false) // nil ж Үи®°
    } else {
        buf.WriteBool(true) // йқһ nil
        buf.WriteInt64(*ov.Value)
    }
}

func (s *OptionalValueSerializer) ReadData(ctx *fory.ReadContext, value reflect.Value) {
    buf := ctx.Buffer()
    err := ctx.Err()
    hasValue := buf.ReadBool(err)
    if !hasValue {
        value.Set(reflect.ValueOf(OptionalValue{Value: nil}))
        return
    }
    v := buf.ReadInt64(err)
    value.Set(reflect.ValueOf(OptionalValue{Value: &v}))
}

func (s *MySerializer) ReadData(ctx *fory.ReadContext, value reflect.Value) {
    buf := ctx.Buffer()
    version := buf.ReadInt8(ctx.Err())

        return

жҢүеҗҚз§°жіЁеҶҢвҖӢ

иҝҷз§Қж–№ејҸжӣҙзҒөжҙ»пјҢдҪҶеәҸеҲ—еҢ–ејҖй”ҖжӣҙеӨ§пјҢеӣ дёәзұ»еһӢеҗҚдјҡеҶҷе…Ҙж•°жҚ®дёӯпјҡ

f.RegisterNamedExtension(MyType{}, "myapp.MyType", &MySerializer{})

жңҖдҪіе®һи·өвҖӢ

  1. дҝқжҢҒж јејҸз®ҖеҚ•пјҢеҸӘеәҸеҲ—еҢ–зңҹжӯЈйңҖиҰҒзҡ„ж•°жҚ®гҖӮ
  2. е°Ҹж•ҙж•°дјҳе…ҲдҪҝз”Ё WriteVarint32 / WriteVarint64гҖӮ
  3. жҢҮй’Ҳе’Ң slice зӯүеҸҜз©әеҖјиҰҒжҳҫејҸеӨ„зҗҶгҖӮ
  4. еҸҜд»ҘиҖғиҷ‘еўһеҠ зүҲжң¬еӯ—иҠӮпјҢдёәе°ҶжқҘж јејҸжј”иҝӣз•ҷеҮәз©әй—ҙгҖӮ
  5. е§Ӣз»ҲйӘҢиҜҒеҫҖиҝ”жӯЈзЎ®жҖ§пјҢеҚі Read(Write(value)) == valueгҖӮ
  6. иҜ»еҸ–йЎәеәҸеҝ…йЎ»дёҺеҶҷе…ҘйЎәеәҸе®Ңе…ЁдёҖиҮҙгҖӮ
  7. иҜ»еҸ–йҳ¶ж®өиҰҒз»“еҗҲ ctx.HasError() еҒҡй”ҷиҜҜжЈҖжҹҘгҖӮ
  8. еңЁеҸ‘йҖҒдҪҝз”ЁиҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁзҡ„ж•°жҚ®еүҚпјҢзЎ®дҝқжүҖжңүжңҚеҠЎйғҪе·Іе®ҢжҲҗеҜ№еә”жіЁеҶҢгҖӮ

жөӢиҜ•иҮӘе®ҡд№үеәҸеҲ—еҢ–еҷЁвҖӢ

func TestMySerializer(t *testing.T) {
    f := fory.New()
    f.RegisterExtension(MyType{}, 100, &MySerializer{})

    original := MyType{Field: "test"}

    // Serialize
    data, err := f.Serialize(original)
    require.NoError(t, err)

    // Deserialize
    var result MyType
    err = f.Deserialize(data, &result)
    require.NoError(t, err)

    assert.Equal(t, original, result)
}
